# 井号开始的为注释行
# 没个数据表按照格式
# 
# 表名 {
#   字段名   字段类型    [PK|+|!|~|UNIQUE] <默认值>[,]
#	...
# }
# 其中：
# +  表示自增长
# !  表示 NOT NULL
# ~  表示 无符号
# 如果 PK，那么 UNIQUE 和  ! 可以省略
#
# 例如:
t_pet {
	id INT +PK,
	name VARCHAR(20),
	masterId
}

t_master {
	id INT +PK,
	name VARCHAR(20)
}

t_food {
	id INT +PK,
	name VARCHAR(20) <BANANA>
}

t_pet_food {
	petId INT,
	foodId INT
}
